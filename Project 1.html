# remove exact and address-based duplicates, reset index
before = len(df)

# basic normalization for matching (won't overwrite original ADDRESS/OWNERNAME)
if 'ADDRESS' in df.columns:
    df['ADDRESS_clean'] = df['ADDRESS'].astype(str).str.upper().str.strip().str.replace(r'\W+', ' ', regex=True).str.replace(r'\s+', ' ', regex=True)
if 'OWNERNAME' in df.columns:
    df['OWNERNAME_clean'] = df['OWNERNAME'].astype(str).str.upper().str.strip().str.replace(r'\W+', ' ', regex=True).str.replace(r'\s+', ' ', regex=True)

# drop exact duplicates (all columns)
df = df.drop_duplicates()
after_exact = len(df)

# drop duplicates by address (and owner if available)
subset = [c for c in ['ADDRESS_clean', 'OWNERNAME_clean'] if c in df.columns]
if subset:
    df = df.drop_duplicates(subset=subset, keep='first')
after_address = len(df)

# remove helper columns and reset index
for c in ['ADDRESS_clean', 'OWNERNAME_clean']:
    if c in df.columns:
        df.drop(columns=[c], inplace=True)
df.reset_index(drop=True, inplace=True)

print(f"rows_before={before}, after_exact={after_exact}, after_address_dedup={after_address}, removed={before-after_address}")
